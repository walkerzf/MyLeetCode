## 题干

> Given an array of integers nums, sort the array in ascending order.
>
>  
>
> Example 1:
>
> Input: nums = [5,2,3,1]
> Output: [1,2,3,5]
> Example 2:
>
> Input: nums = [5,1,1,2,0,0]
> Output: [0,0,1,1,2,5]
>
>
> Constraints:
>
> 1 <= nums.length <= 50000
> -50000 <= nums[i] <= 50000

## Solution Buddle

```java
//TLE
class Solution {
    public List<Integer> sortArray(int[] nums) {
         for(int i=nums.length-1;i>0;i--){
             int flag=0;
             for(int j=1;j<=i;j++){
                 if(nums[j]<nums[j-1]){
                     int tmp=nums[j-1];
                     nums[j-1]=nums[j];
                     nums[j]=tmp;
                     flag=1;
                 }
             }
             if(flag==0) break;
         }
         List<Integer> res=new LinkedList<>();
         for(int i=0;i<nums.length;i++){
                res.add(nums[i]);
         }
         return res;
    }
}
```

## Merge Sort

```java
//time：12ms 48.2MB
class Solution {
    public List<Integer> sortArray(int[] nums) {
         mergeSort(nums,0,nums.length-1);
         List<Integer> res=new LinkedList<>();
         for(int i=0;i<nums.length;i++){
                res.add(nums[i]);
         }
         return res;
    }

    private void mergeSort(int []nums,int left,int right){
        if(left==right){
            return;
        }
        int mid=left+(right-left)/2;
        mergeSort(nums,left,mid);
        mergeSort(nums,mid+1,right);
        MergeSortInOrder(nums,left,mid,right);
    }

    private void MergeSortInOrder(int []nums,int left,int mid,int right){
        int [] tmp=new int[right-left+1];
        int i=left;int j=mid+1;
        int k=0;
        while(i<=mid&&j<=right){
            if(nums[i]<nums[j]) tmp[k++]=nums[i++];
            else tmp[k++]=nums[j++];
        }
        while(i<=mid) tmp[k++]=nums[i++];
        while(j<=right) tmp[k++]=nums[j++];
        int m=0;
        for(int l=left;l<=right;l++){
            nums[l]=tmp[m++];
        }
    }
}
```

## Quick Sort

```java
//9ms 48.8MB
class Solution {
    public List<Integer> sortArray(int[] nums) {
         quickSort(nums,0,nums.length-1);
         List<Integer> res=new LinkedList<>();
         for(int i=0;i<nums.length;i++){
                res.add(nums[i]);
         }
         return res;
    }
    private void quickSort(int []nums,int left,int right){
        int l=left;
        int r=right;
        if(left<right){
            int pivot=nums[left]; //pivot
            while(left<right){
                while(left<right&&nums[right]>pivot) right--;
                if(left<right) nums[left++]=nums[right];
                while(left<right&&nums[left]<pivot) left++;
                if(left<right) nums[right--]=nums[left];
            }   
            nums[left]=pivot;
            quickSort(nums,l,left-1);
            quickSort(nums,left+1,r);
        }
        return;
    }
}
```

```java
class Solution {
```

```java
         int number[]=new int[2*50000+1];
         for(int i=0;i<nums.length;i++){

         }

                number[i]--;
            }

         return res;
```

## Base Sort 基数排序

```java
//time 15ms 52.3MB
class Solution {
    public List<Integer> sortArray(int[] nums) {
         int bk[]=new int[19];
         int max=Integer.MIN_VALUE;
         for(int i=0;i<nums.length;i++){
             if(max<Math.abs(nums[i])) max=nums[i];
         }
         if(max<0) max=-max;
         int count= Integer.toString(max).length();
         int bd[][]=new int [19][nums.length];
         for(int i=0;i<count;i++){
             for(int j=0;j<nums.length;j++){
                 int val=(int)((nums[j]/Math.pow(10,i))%10);
                 bd[val+9][bk[val+9]++]=nums[j];
             }
             int index=0;
             for(int hi=0;hi<19;hi++){
                 if(bk[hi]!=0){
                 for(int l=0;l<bk[hi];l++){
                     nums[index++]=bd[hi][l];
                    }
                 }
             }

            index=0;
            bk=new int[19];
         }
         

         List<Integer> res=new LinkedList<>();
         for(int i=0;i<nums.length;i++){
                res.add(nums[i]);
         }
         return res;

    }
}
```
